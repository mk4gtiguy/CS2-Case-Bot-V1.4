using System;
using System.Collections.Generic;

public class CPHInline
{
    public bool Execute()
    {
        var random = new Random();
        
        // Get user and case
        string user = CPH.GetGlobalVar<string>("User");
        string currentCase = CPH.GetGlobalVar<string>("CurrentCase");
        string casePrefix = currentCase.Replace(" ", "").Replace("&", "").Replace("-", "");
        
        // Check streak
        int currentStreak = CPH.GetGlobalVar<int>(user + "_OpenStreak");
        currentStreak++;
        CPH.SetGlobalVar(user + "_OpenStreak", currentStreak, true);
        
        // Check for streak bonuses
        if (currentStreak == 10 || currentStreak == 25 || currentStreak == 50 || currentStreak == 100)
        {
            double bonus = 0;
            if (currentStreak == 10) bonus = 5.00;
            else if (currentStreak == 25) bonus = 15.00;
            else if (currentStreak == 50) bonus = 50.00;
            else if (currentStreak == 100) bonus = 200.00;
            
            double balance = CPH.GetGlobalVar<double>(user + "_Balance");
            balance += bonus;
            CPH.SetGlobalVar(user + "_Balance", balance, true);
            CPH.SendMessage("🔥 " + user + " reached a " + currentStreak + " open streak! Bonus $" + bonus.ToString("F2") + " added!");
        }
        
        // Roll rarity
        int roll = random.Next(1, 1001);
        string rarityType;
        if (roll <= 26) rarityType = "Gold";
        else if (roll >= 975) rarityType = "Red";
        else if (roll >= 950) rarityType = "Pink";
        else if (roll >= 900) rarityType = "Purple";
        else rarityType = "Blue";
        
        // Base values
        string emoji;
        double baseValue;
        if (rarityType == "Gold") { emoji = "🟡"; baseValue = 50.00; }
        else if (rarityType == "Red") { emoji = "🔴"; baseValue = 5.00; }
        else if (rarityType == "Pink") { emoji = "💗"; baseValue = 1.50; }
        else if (rarityType == "Purple") { emoji = "🟣"; baseValue = 0.50; }
        else { emoji = "🔷"; baseValue = 0.10; }
        
        // Get skin
        string skinList = CPH.GetGlobalVar<string>(casePrefix + rarityType + "List");
        if (string.IsNullOrEmpty(skinList)) skinList = CPH.GetGlobalVar<string>(rarityType + "List");
        
        var validSkins = new List<string>();
        foreach (string skin in skinList.Split(';'))
            if (!string.IsNullOrWhiteSpace(skin))
                validSkins.Add(skin.Trim());
        
        string currentSkin = validSkins[random.Next(0, validSkins.Count)];
        
        // Apply condition (FN, MW, FT, WW, BS)
        string[] conditions = { "Factory New", "Minimal Wear", "Field-Tested", "Well-Worn", "Battle-Scarred" };
        double[] conditionMultipliers = { 2.0, 1.5, 1.0, 0.75, 0.5 };
        int conditionRoll = random.Next(1, 101);
        int conditionIndex = 0;
        if (conditionRoll <= 5) conditionIndex = 0;
        else if (conditionRoll <= 20) conditionIndex = 1;
        else if (conditionRoll <= 70) conditionIndex = 2;
        else if (conditionRoll <= 90) conditionIndex = 3;
        else conditionIndex = 4;
        
        string condition = conditions[conditionIndex];
        double finalValue = baseValue * conditionMultipliers[conditionIndex];
        
        // Apply StatTrak (10% chance)
        bool isStatTrak = random.Next(1, 101) <= 10;
        if (isStatTrak)
        {
            finalValue *= 2;
        }
        
        string statTrakTag = isStatTrak ? " (StatTrak™)" : "";
        string newItem = emoji + " " + currentSkin + " (" + condition + ")" + statTrakTag;
        
        // Save to inventory
        string oldItems = CPH.GetGlobalVar<string>(user + "_Items");
        string newItemsList = string.IsNullOrEmpty(oldItems) ? newItem : oldItems + ";" + newItem;
        CPH.SetGlobalVar(user + "_Items", newItemsList, true);
        
        // Track case opens
        string caseKey = casePrefix + "_Opened";
        int caseOpened = CPH.GetGlobalVar<int>(caseKey);
        CPH.SetGlobalVar(caseKey, caseOpened + 1, true);
        
        int userCaseOpened = CPH.GetGlobalVar<int>(user + "_" + casePrefix + "_Opened");
        CPH.SetGlobalVar(user + "_" + casePrefix + "_Opened", userCaseOpened + 1, true);
        
        int totalOpened = CPH.GetGlobalVar<int>(user + "_TotalOpened");
        CPH.SetGlobalVar(user + "_TotalOpened", totalOpened + 1, true);
        
        // Check case unlocks (every 25 opens unlocks a new case)
        string unlockedCases = CPH.GetGlobalVar<string>(user + "_UnlockedCases");
        if (string.IsNullOrEmpty(unlockedCases)) unlockedCases = "Dreams;Recoil";
        
        int totalCasesOpened = totalOpened + 1;
        if (totalCasesOpened >= 25 && !unlockedCases.Contains("Kilowatt"))
        {
            unlockedCases += ";Kilowatt";
            CPH.SendMessage("🔓 " + user + " unlocked the Kilowatt case! Type !setcase Kilowatt to try it!");
        }
        if (totalCasesOpened >= 50 && !unlockedCases.Contains("Revolution"))
        {
            unlockedCases += ";Revolution";
            CPH.SendMessage("🔓 " + user + " unlocked the Revolution case!");
        }
        if (totalCasesOpened >= 75 && !unlockedCases.Contains("Fracture"))
        {
            unlockedCases += ";Fracture";
            CPH.SendMessage("🔓 " + user + " unlocked the Fracture case!");
        }
        if (totalCasesOpened >= 100 && !unlockedCases.Contains("Clutch"))
        {
            unlockedCases += ";Clutch";
            CPH.SendMessage("🔓 " + user + " unlocked the Clutch case!");
        }
        if (totalCasesOpened >= 150 && !unlockedCases.Contains("Prisma"))
        {
            unlockedCases += ";Prisma";
            CPH.SendMessage("🔓 " + user + " unlocked the Prisma case!");
        }
        if (totalCasesOpened >= 200 && !unlockedCases.Contains("DangerZone"))
        {
            unlockedCases += ";DangerZone";
            CPH.SendMessage("🔓 " + user + " unlocked the Danger Zone case!");
        }
        CPH.SetGlobalVar(user + "_UnlockedCases", unlockedCases, true);
        
        // Track rarity counts
        string rarityCounterKey = user + "_" + rarityType + "s";
        int rarityCount = CPH.GetGlobalVar<int>(rarityCounterKey);
        CPH.SetGlobalVar(rarityCounterKey, rarityCount + 1, true);
        
        // Check achievements
        string achievements = CPH.GetGlobalVar<string>(user + "_Achievements");
        if (string.IsNullOrEmpty(achievements)) achievements = "";
        
        if (rarityType == "Gold" && !achievements.Contains("FirstGold"))
        {
            achievements += (achievements == "" ? "" : ";") + "FirstGold";
            CPH.SendMessage("🏆 " + user + " earned the 'First Gold' achievement!");
        }
        
        int tradeupCount = CPH.GetGlobalVar<int>(user + "_TradeupCount");
        if (tradeupCount >= 10 && !achievements.Contains("TradeUpMaster"))
        {
            achievements += (achievements == "" ? "" : ";") + "TradeUpMaster";
            CPH.SendMessage("🏆 " + user + " earned the 'Trade Up Master' achievement!");
        }
        
        int jackpotWins = CPH.GetGlobalVar<int>(user + "_JackpotWins");
        if (jackpotWins >= 1 && !achievements.Contains("JackpotWinner"))
        {
            achievements += (achievements == "" ? "" : ";") + "JackpotWinner";
            CPH.SendMessage("🏆 " + user + " earned the 'Jackpot Winner' achievement!");
        }
        
        // Check if they've opened all case types
        string[] allCases = { "Dreams", "Recoil", "Kilowatt", "Revolution", "Fracture", "Clutch", "Prisma", "DangerZone", "Horizon", "Chroma3", "Spectrum2" };
        int openedCount = 0;
        foreach (string c in allCases)
        {
            int count = CPH.GetGlobalVar<int>(user + "_" + c.Replace(" ", "") + "_Opened");
            if (count > 0) openedCount++;
        }
        if (openedCount >= 6 && !achievements.Contains("CaseCollector"))
        {
            achievements += (achievements == "" ? "" : ";") + "CaseCollector";
            CPH.SendMessage("🏆 " + user + " earned the 'Case Collector' achievement!");
        }
        
        CPH.SetGlobalVar(user + "_Achievements", achievements, true);
        
        // Gold tracking
        if (rarityType == "Gold")
        {
            int userGolds = CPH.GetGlobalVar<int>(user + "_Golds");
            CPH.SetGlobalVar(user + "_Golds", userGolds + 1, true);
            int totalGolds = CPH.GetGlobalVar<int>("GoldPulls_Total");
            CPH.SetGlobalVar("GoldPulls_Total", totalGolds + 1, true);
        }
        
        // Track all users
        string allUsers = CPH.GetGlobalVar<string>("AllUsers");
        if (string.IsNullOrEmpty(allUsers)) allUsers = user;
        else
        {
            bool userExists = false;
            foreach (string existingUser in allUsers.Split('§'))
                if (existingUser.Equals(user, StringComparison.OrdinalIgnoreCase))
                { userExists = true; break; }
            if (!userExists) allUsers += "§" + user;
        }
        CPH.SetGlobalVar("AllUsers", allUsers, true);
        
        CPH.SetGlobalVar("CurrentSkin", currentSkin, true);
        CPH.SetGlobalVar("RarityType", rarityType, true);
        CPH.SetGlobalVar("Emoji", emoji, true);
        CPH.SetGlobalVar("Value", finalValue.ToString("F2"), true);
        CPH.SetGlobalVar("CurrentValue", finalValue, true);
        
        // DELAYED OPENING - 2.5 second wait for suspense
        System.Threading.Thread.Sleep(2500);
        
        CPH.SendMessage("🎁 " + user + " opened " + currentCase + " and got: " + newItem + " (Value: $" + finalValue.ToString("F2") + ")");
        return true;
    }
}

using System;

public class CPHInline
{
    public bool Execute()
    {
        string user = args["userName"].ToString();
        string rawInput = args["rawInput"].ToString();
        
        string caseName = rawInput.Replace("!setcase", "").Trim();
        
        // All available cases
        string[] allCases = { "Dreams", "Recoil", "Kilowatt", "Revolution", "Fracture", "Clutch", "Prisma", "DangerZone", "Horizon", "Chroma3", "Spectrum2" };
        bool isValid = false;
        
        foreach (string validCase in allCases)
        {
            if (caseName.Equals(validCase, StringComparison.OrdinalIgnoreCase))
            {
                caseName = validCase;
                isValid = true;
                break;
            }
        }
        
        if (!isValid)
        {
            CPH.SendMessage("📦 " + user + " Unknown case: '" + caseName + "' | Available: Dreams, Recoil, Kilowatt, Revolution, Fracture, Clutch, Prisma, DangerZone, Horizon, Chroma3, Spectrum2");
            return false;
        }
        
        // Check if unlocked
        string unlockedCases = CPH.GetGlobalVar<string>(user + "_UnlockedCases");
        if (string.IsNullOrEmpty(unlockedCases)) unlockedCases = "Dreams;Recoil";
        
        if (!unlockedCases.Contains(caseName) && caseName != "Dreams" && caseName != "Recoil")
        {
            CPH.SendMessage("🔒 " + user + " you haven't unlocked " + caseName + " yet! Keep opening cases to unlock more!");
            return false;
        }
        
        CPH.SetGlobalVar(user + "_LastCase", caseName, true);
        
        // Get cost for display
        double cost = 1.50;
        if (caseName.Equals("Recoil", StringComparison.OrdinalIgnoreCase)) cost = 0.50;
        else if (caseName.Equals("Fracture", StringComparison.OrdinalIgnoreCase) || 
                 caseName.Equals("Kilowatt", StringComparison.OrdinalIgnoreCase)) cost = 2.00;
        else if (caseName.Equals("Clutch", StringComparison.OrdinalIgnoreCase)) cost = 3.00;
        else if (caseName.Equals("Prisma", StringComparison.OrdinalIgnoreCase)) cost = 2.50;
        else if (caseName.Equals("DangerZone", StringComparison.OrdinalIgnoreCase)) cost = 2.50;
        else if (caseName.Equals("Horizon", StringComparison.OrdinalIgnoreCase)) cost = 2.00;
        else if (caseName.Equals("Chroma3", StringComparison.OrdinalIgnoreCase)) cost = 2.00;
        else if (caseName.Equals("Spectrum2", StringComparison.OrdinalIgnoreCase)) cost = 2.50;
        
        CPH.SendMessage("📦 " + user + " switched to " + caseName + " case! ($" + cost.ToString("F2") + " per open) | Type !case to open it.");
        return true;
    }

using System;

public class CPHInline
{
    public bool Execute()
    {
        string user = args["userName"].ToString();
        string currentCase = CPH.GetGlobalVar<string>(user + "_LastCase");
        
        if (string.IsNullOrEmpty(currentCase))
        {
            CPH.SendMessage(user + " please set a case first with !setcase Dreams (or Recoil, Kilowatt, etc.)");
            return false;
        }
        
        CPH.SetGlobalVar("User", user, true);
        CPH.SetGlobalVar("CurrentCase", currentCase, true);
        
        // Get case prefix for cost calculation
        string casePrefix = currentCase.Replace(" ", "").Replace("&", "").Replace("-", "");
        
        // Determine cost (check for custom price first)
        double customPrice = CPH.GetGlobalVar<double>("Price_" + currentCase);
        double cost = customPrice > 0 ? customPrice : 1.50;
        
        if (casePrefix.Equals("Recoil", StringComparison.OrdinalIgnoreCase)) cost = 0.50;
        else if (casePrefix.Equals("Fracture", StringComparison.OrdinalIgnoreCase) || 
                 casePrefix.Equals("Kilowatt", StringComparison.OrdinalIgnoreCase)) cost = 2.00;
        else if (casePrefix.Equals("Clutch", StringComparison.OrdinalIgnoreCase)) cost = 3.00;
        else if (casePrefix.Equals("Prisma", StringComparison.OrdinalIgnoreCase)) cost = 2.50;
        else if (casePrefix.Equals("DangerZone", StringComparison.OrdinalIgnoreCase)) cost = 2.50;
        else if (casePrefix.Equals("Horizon", StringComparison.OrdinalIgnoreCase)) cost = 2.00;
        else if (casePrefix.Equals("Chroma3", StringComparison.OrdinalIgnoreCase)) cost = 2.00;
        else if (casePrefix.Equals("Spectrum2", StringComparison.OrdinalIgnoreCase)) cost = 2.50;
        
        // Check balance
        double balance = CPH.GetGlobalVar<double>(user + "_Balance");
        
        if (balance < cost)
        {
            double shortfall = cost - balance;
            CPH.SendMessage(user + " you need $" + cost.ToString("F2") + " to open " + currentCase + " but only have $" + balance.ToString("F2") + " — you're $" + shortfall.ToString("F2") + " short! Try !daily or !sell.");
            return false;
        }
        
        // Deduct cost
        balance -= cost;
        CPH.SetGlobalVar(user + "_Balance", balance, true);
        CPH.SetGlobalVar("CaseCost", cost, true);
        
        // Show purchase confirmation
        CPH.SendMessage("🔑 " + user + " bought a " + currentCase + " key for $" + cost.ToString("F2") + " | Balance: $" + balance.ToString("F2"));
        
        return true;
    }
}

using System;

public class CPHInline
{
    public bool Execute()
    {
        string user = args["userName"].ToString();
        string items = CPH.GetGlobalVar<string>(user + "_Items");

        if (string.IsNullOrEmpty(items))
        {
            CPH.SendMessage(user + " Your inventory is empty! Type !setcase Dreams then !case to open a case!");
            return true;
        }

        string[] itemList = items.Split(';');
        
        int countGold = 0, countRed = 0, countPink = 0, countPurple = 0, countBlue = 0;
        double totalValue = 0;
        
        foreach (string item in itemList)
        {
            if (item.Contains("🟡")) { countGold++; totalValue += 50.00; }
            else if (item.Contains("🔴")) { countRed++; totalValue += 5.00; }
            else if (item.Contains("💗")) { countPink++; totalValue += 1.50; }
            else if (item.Contains("🟣")) { countPurple++; totalValue += 0.50; }
            else if (item.Contains("🔷")) { countBlue++; totalValue += 0.10; }
            
            // Apply condition multipliers for value calculation
            if (item.Contains("Factory New")) totalValue += totalValue;
            else if (item.Contains("Minimal Wear")) totalValue += totalValue / 2;
            else if (item.Contains("Well-Worn")) totalValue -= totalValue / 4;
            else if (item.Contains("Battle-Scarred")) totalValue -= totalValue / 2;
            
            // StatTrak
            if (item.Contains("StatTrak")) totalValue += totalValue;
        }
        
        // Show first few items as examples with numbers
        string exampleItems = "";
        for (int i = 0; i < Math.Min(3, itemList.Length); i++)
        {
            string shortItem = itemList[i].Length > 25 ? itemList[i].Substring(0, 22) + "..." : itemList[i];
            exampleItems += (i + 1) + "." + shortItem + " ";
        }
        
        CPH.SendMessage("📦 " + user + "'s Inventory — 🟡 " + countGold + " | 🔴 " + countRed + " | 💗 " + countPink + " | 🟣 " + countPurple + " | 🔷 " + countBlue + " | 💰 $" + totalValue.ToString("F2") + " | Items: " + exampleItems + " | Type !sell to sell items by number!");
        return true;
    }

using System;
using System.Collections.Generic;

public class CPHInline
{
    public bool Execute()
    {
        string user = args["userName"].ToString();
        string rawInput = args["rawInput"].ToString();
        string items = CPH.GetGlobalVar<string>(user + "_Items");

        if (string.IsNullOrEmpty(items))
        {
            CPH.SendMessage(user + " Your inventory is empty! Open some cases first with !setcase then !case");
            return false;
        }

        string[] itemList = items.Split(';');
        
        // Parse which item to sell
        string[] parts = rawInput.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        
        int sellIndex = -1;
        
        // If no number provided, show inventory with numbers
        if (parts.Length < 2)
        {
            // Show numbered inventory
            string inventoryList = "";
            for (int i = 0; i < itemList.Length && i < 10; i++)
            {
                // Truncate long item names for cleaner display
                string shortItem = itemList[i].Length > 40 ? itemList[i].Substring(0, 37) + "..." : itemList[i];
                inventoryList += (i + 1) + ". " + shortItem + " | ";
            }
            if (itemList.Length > 10) inventoryList += "... and " + (itemList.Length - 10) + " more";
            
            CPH.SendMessage("📦 " + user + " Type !sell [number] to sell an item. Example: !sell 2 | Your items: " + inventoryList);
            return false;
        }
        
        // Parse the number
        if (!int.TryParse(parts[1], out sellIndex))
        {
            CPH.SendMessage(user + " Please use a number! Example: !sell 3 | Type !sell alone to see your numbered inventory.");
            return false;
        }
        
        // Adjust for 1-based indexing
        sellIndex--;
        
        if (sellIndex < 0 || sellIndex >= itemList.Length)
        {
            CPH.SendMessage(user + " Invalid item number! You have " + itemList.Length + " items. Type !sell to see them numbered 1-" + itemList.Length);
            return false;
        }
        
        string soldItem = itemList[sellIndex].Trim();

        // Calculate value with condition and StatTrak multipliers
        double value = 0.10; // default Blue value
        
        // Base value by rarity
        if (soldItem.Contains("🟡")) value = 50.00;
        else if (soldItem.Contains("🔴")) value = 5.00;
        else if (soldItem.Contains("💗")) value = 1.50;
        else if (soldItem.Contains("🟣")) value = 0.50;
        else value = 0.10;
        
        // Check for condition multipliers
        if (soldItem.Contains("Factory New")) value *= 2.0;
        else if (soldItem.Contains("Minimal Wear")) value *= 1.5;
        else if (soldItem.Contains("Well-Worn")) value *= 0.75;
        else if (soldItem.Contains("Battle-Scarred")) value *= 0.5;
        // Field-Tested is 1.0x, no change
        
        // Check for StatTrak
        if (soldItem.Contains("StatTrak")) value *= 2;
        
        // Remove the item
        var newList = new List<string>(itemList);
        newList.RemoveAt(sellIndex);
        CPH.SetGlobalVar(user + "_Items", string.Join(";", newList), true);

        // Add to balance
        double balance = CPH.GetGlobalVar<double>(user + "_Balance");
        balance += value;
        CPH.SetGlobalVar(user + "_Balance", balance, true);
        
        // Reset streak on sell
        CPH.SetGlobalVar(user + "_OpenStreak", 0, true);
        
        // Show remaining item count
        int remainingCount = newList.Count;
        
        CPH.SendMessage("💰 " + user + " sold " + soldItem + " for $" + value.ToString("F2") + " | New balance: $" + balance.ToString("F2") + " | " + remainingCount + " items left in inventory | Your open streak was reset!");
        return true;
    }
}
